Length

Calculates the length of a given string.
def length(string): count = 0 for _ in string: count += 1 return count # Example usage text = "Hello" result = length(text) print(f"Length of '{text}' is {result}")
Output: Length of 'Hello' is 5

Concatenation

Joins two strings together.
def concat(str1, str2): return str1 + str2 # Example usage text1 = "Auto" text2 = "mata" result = concat(text1, text2) print(f"Concatenation: {result}")
Output: Concatenation: Automata

Prefix

Finds all prefixes of a given string.
def get_prefixes(string): prefixes = [] for i in range(len(string) + 1): prefixes.append(string[:i]) return prefixes # Example usage text = "auto" result = get_prefixes(text) print(f"Prefixes of '{text}': {result}")
Output: Prefixes of 'auto': ['', 'a', 'au', 'aut', 'auto']

Suffix

Finds all suffixes of a given string.
def get_suffixes(string): suffixes = [] for i in range(len(string) + 1): suffixes.append(string[i:]) return suffixes # Example usage text = "auto" result = get_suffixes(text) print(f"Suffixes of '{text}': {result}")
Output: Suffixes of 'auto': ['auto', 'uto', 'to', 'o', '']

Reverse

Reverses the characters in a string.
def reverse(string): return string[::-1] # Example usage text = "automata" result = reverse(text) print(f"Reverse of '{text}': {result}")
Output: Reverse of 'automata': atamotua

Palindrome

Checks if a string reads the same forwards and backwards.
def is_palindrome(string): return string == string[::-1] # Example usage text1 = "radar" text2 = "auto" print(f"Is '{text1}' palindrome? {is_palindrome(text1)}") print(f"Is '{text2}' palindrome? {is_palindrome(text2)}")
Output: Is 'radar' palindrome? True Is 'auto' palindrome? False

Union

Combines elements from two sets, removing duplicates.
def union(set1, set2): return list(set(set1) | set(set2)) # Example usage set1 = ['a', 'b', 'c'] set2 = ['b', 'c', 'd'] result = union(set1, set2) print(f"Union of {set1} and {set2}: {result}")
Output: Union of ['a', 'b', 'c'] and ['b', 'c', 'd']: ['a', 'b', 'c', 'd']

Intersection

Finds common elements between two sets.
def intersection(set1, set2): return list(set(set1) & set(set2)) # Example usage set1 = ['a', 'b', 'c'] set2 = ['b', 'c', 'd'] result = intersection(set1, set2) print(f"Intersection of {set1} and {set2}: {result}")
Output: Intersection of ['a', 'b', 'c'] and ['b', 'c', 'd']: ['b', 'c']

Difference

Finds elements in first set that are not in second set.
def difference(set1, set2): return list(set(set1) - set(set2)) # Example usage set1 = ['a', 'b', 'c'] set2 = ['b', 'c', 'd'] result = difference(set1, set2) print(f"Difference of {set1} - {set2}: {result}")
Output: Difference of ['a', 'b', 'c'] - ['b', 'c', 'd']: ['a']

Symmetric Difference

Finds elements that are in either set, but not in both.
def symmetric_difference(set1, set2): return list(set(set1) ^ set(set2)) # Example usage set1 = ['a', 'b', 'c'] set2 = ['b', 'c', 'd'] result = symmetric_difference(set1, set2) print(f"Symmetric difference of {set1} and {set2}: {result}")
Output: Symmetric difference of ['a', 'b', 'c'] and ['b', 'c', 'd']: ['a', 'd']

Powerset

Finds all possible subsets of a set.
def powerset(s): x = len(s) result = [] for i in range(1 << x): subset = [s[j] for j in range(x) if (i & (1 << j))] result.append(subset) return result # Example usage set1 = ['a', 'b', 'c'] result = powerset(set1) print(f"Powerset of {set1}: {result}")
Output: Powerset of ['a', 'b', 'c']: [[], ['a'], ['b'], ['a', 'b'], ['c'], ['a', 'c'], ['b', 'c'], ['a', 'b', 'c']]
